#include <iostream>

using namespace std;

void troca_errado( double a, double b ){ // n troca pq são cópias (pra pegar esses valores precisaria de um return e alterar valores fora da func)
    double copo = a;
    a = b;
    b = copo;
}

void troca_certo( double& a, double& b ){ // usa referência
    double copo = a;
    a = b;
    b = copo;
}

void troca_errado2( const double& a, const double& b ){ // dá erro pq tenta mudar valor de const
    double copo = a;
    a = b;
    b = copo;
}
